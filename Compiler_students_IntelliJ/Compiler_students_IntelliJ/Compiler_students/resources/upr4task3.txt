program{

    factorial(int n )-> int{
        int result = 1;
        int i = 1;

        while (i <= n) {
            result =result * i;
            i=i+1;

        }
        return result;

    }

    main()->void{

        int num = 5;
        print("Iterative factorial of "+ num+" is: ",@factorial(num);


    }



}
